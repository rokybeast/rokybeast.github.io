PolyScript Language

A modern, expressive programming language designed for clarity, power, and elegant code. Explore the features that make PolyScript unique.


Show Full Syntax

Type Safety & Modifiers

Explicit variable declarations with powerful modifiers like !isConstant, !invisible, and !REQUIRE for fine-grained control over your data.

Advanced Import System

Flexible imports with aliasing support. Import entire modules or specific functions with clean, readable syntax.

Fragments & Composition

Compose functionality with fragments - a unique approach to code organization that's more flexible than traditional inheritance.

Decorators & Methods

Enhance your functions with decorators and create reusable method patterns for clean, maintainable code.

Smart Collections

Arrays, tuples, and collections with position-based operations and intelligent indexing systems.

Async by Design

Built-in async support with threadTask and real-time execution capabilities for modern applications.

Language Features in Action

// Variables with Modifiers
Option.ExplicitVariables = true;
new gravity as float: 9.8 !isConstant;
new apiKey as string: "secret123" !invisible;

// Advanced Import System  
import MathLib as 'Math';
import { sqrt: 'SquareRoot', pow: 'Power' } from MathLib;
// Smart Collections & Arrays
new values as array[int];
values.new(5 : 1);    // Insert 5 at position 1
values.new(90 : 0);   // Insert 90 at position 0
values.delete(5 : any); // Delete 5 from any position

new fruits as collection: {
    "apple": 1,
    "banana": 2,
    "kiwi": 3
};
write(fruits["apple"]); // Access by key
write(fruits[3]);       // Access by position
// Decorators and Custom Methods
@logExecution
public function calculate(param: int) {
    return param * 2;
}

new method States(){
    hoverState(apply): {
        gui.cursor.onSelObject();
    } -> {
        self.color.darken(+5);
    }
    
    clickState(apply): {
        gui.cursor.onSelObject(isClicked);
    } -> {
        continue;
    }
}
// Fragments - Composition over Inheritance
fragment UserProfile {
    username: string;
    email: string;
    
    function displayProfile() {
        write("User:", username, "Email:", email);
    }
}

fragment UserSettings {
    theme: string;
    notifications: boolean;
}

// Compose fragments
new UserFullProfile as fragment {
    !Append UserProfile;
    !Append UserSettings;
    
    function displayFullProfile() {
        self.displayProfile();
        self.displaySettings();
    }
}
// Pattern Matching & Control Flow
match userInput {
    when 1: write("Option One");
    when int n if n > 5: write("Greater than five:", n);
    when string s if s.length > 0: write("Non-empty string:", s);
    else: write("Unknown input");
}

// Modern Operators
new username = inputName ~> "Guest";  // Elvis operator
new result = (score > 50) |> "Pass" : "Fail";  // Ternary
// Async & Threading
async function fetchUserData(userId: int): object {
    new userData as object = await apiCall("/users/" + userId);
    await sleep(100); // Non-blocking delay
    return userData;
}

threadTask !realTime: function processData() {
    new users as array = await fetchUserData(123);
    write("Processing complete:", users.length, "users");
}
// Error Handling with Context
try {
    new score as int = read.in("Enter score: ");
    if (score < 0) throw new RangeError("Score cannot be negative.");
    if (score > 100) Error.Raise("Invalid range.", 400, "SCORE_ERR");
} catch (err -> e) {
    write("Validation failed:", e.message);
} finally {
    write("Score validation complete.");
}
